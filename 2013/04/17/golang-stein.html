Go——Stein算法求最大公约数

18 Apr 2013

package main

func gcd(p, q int) int {
    if p < q {
        t := p
        p = q
        q = t
    }
    if q == 0 {
        return p
    }
    if p&1 == 0 && q&1 == 0 {
        return gcd(p>>1, q>>1) << 1
    }
    if p&1 == 0 {
        return gcd(p>>1, q)
    }
    if q&1 == 0 {
        return gcd(p, q>>1)
    }
    return gcd((p+q)>>1, (p-q)>>1)
    }

    func main() {
    p, q := 3, 11
    println(gcd(p, q))
}
